# nvinfer config for MediaPipe Hand Landmark (PINTO0309/hand_landmark
# v1.0.0, `hand_landmark_sparse_Nx3x224x224.onnx`).
#
# Runs as a TERTIARY GIE on each `hand` child meta produced by the
# YOLOX-BHH secondary (gie-unique-id=3, class_id=2). For every hand
# crop the model emits:
#   xyz_x21                    : 63 floats — 21 keypoints x (x, y, z)
#                                with x/y in 0..224 input-pixel coords
#                                and z as relative depth.
#   hand_score                 :  1 float — presence/confidence.
#   lefthand_0_or_righthand_1  :  1 float — handedness (0=L, 1=R).
#
# All postprocessing happens in the Python pad probe in app.py:
# project keypoints from 224x224 model space → hand bbox image space
# and draw the 21-point skeleton via NvDsDisplayMeta. `network-type=100`
# disables nvinfer's built-in classifier/detector parsing (neither shape
# fits), `output-tensor-meta=1` attaches the raw tensors to the hand
# object's user-meta list.
[property]
gpu-id=0
# MediaPipe expects 0..1 normalized floats, RGB. 1/255 = 0.0039215686.
net-scale-factor=0.0039215686
offsets=0;0;0
model-color-format=0

onnx-file=/var/lib/nvidia-deepstream/models/handlandmark/hand_landmark_sparse_224x224.onnx
model-engine-file=/var/lib/nvidia-deepstream/models/handlandmark/hand_landmark_sparse_224x224.onnx_b1_gpu0_fp16.engine

# The ONNX has dynamic batch (N). Pin to 1 here — multi-hand pose runs
# sequentially over hands within a frame, which is fine for the live
# webcam use case.
batch-size=1
network-mode=2
# `Other` — read the raw tensor in our pad probe.
network-type=100
output-tensor-meta=1
# Process each YOLOX-BHH `hand` detection (class_id=2 in handdet).
process-mode=2
operate-on-gie-id=3
operate-on-class-ids=2
gie-unique-id=4

# Don't run landmark inference on tiny hand crops — keypoints become
# noise below ~24 px.
input-object-min-width=24
input-object-min-height=24

interval=0
